```js
// PATCH 2026-07-10 (Slot-Override-Refactor v2)
if (slotDef.slot_override_erlaubt === true && filterStr && filterStr !== '') {
  const filters = filterStr.split('|').map(s => s.trim()).filter(Boolean);
  const additionalCandidates = pool.filter(p => {
    if (candidates.includes(p)) return false;
    const hfSet = new Set(csvToArr(p.hauptfunktion));
    const nfSet = new Set(csvToArr(p.nebenfunktionen));
    return filters.some(fStr => {
      const flagMatch = fStr.match(/^([a-zA-Z_][a-zA-Z0-9_]*)=(TRUE|FALSE)$/);
      if (flagMatch) return boolVal(p[flagMatch[1]]) === (flagMatch[2] === 'TRUE');
      return p.produkt_key === fStr || p.produktlinie === fStr
          || hfSet.has(fStr) || nfSet.has(fStr);
    });
  });
  if (additionalCandidates.length > 0) candidates = candidates.concat(additionalCandidates);
}
```

## Neue Konventionen

- **Vor-Simulation muss HF UND NF prüfen** (nicht nur HF wie in der 07-09-Session). Neue Engine-Semantik ist symmetrisch, Vor-Simulation muss es auch sein.
- **Bool-Flag als whitelabel-präziser Ausweg**: Wenn ein Wirkungs-Token zu viele Produkte matched (Ranking-Zufall) und Produkt-Key-Filter nicht whitelabel-tauglich ist, ist ein exklusiver Bool-Flag der pragmatische Weg. Semantische Enge des Flags ist der Preis für Qualitäts- + Whitelabel-Sicherheit.
- **Grid-Resize vor neuer Spalte**: `ws.add_cols(1)` **vor** `update_cell(1, new_col, header)`, sonst `APIError: Range exceeds grid limits`.
